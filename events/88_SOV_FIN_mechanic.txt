	country_event = {
		id = sovfin_events.1
		hidden = yes

		is_triggered_only = yes
		
		immediate = { 
			hidden_effect = {
				if = {
					limit = { 
						casualties_k > 150 
						date < 1941.5.5
					}    
					SOV = { set_country_flag = SOV_war_since_purge}
					SOV = {
						country_event = { id = finland.8 days = 7 }
						white_peace = FIN
					}
					GER = { add_to_faction = FIN }
				}
				else = { 
					FIN = { 
						country_event = { id = sovfin_events.2 days = 3 } 
					} 
				}
			}
		}	
	}
